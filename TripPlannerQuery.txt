CREATE TABLE UserID(
ID INT NOT NULL PRIMARY KEY IDENTITY (1,1),
PhoneNumber INT,
Name NVARCHAR (50),
DOB DATE,
Address NVARCHAR(50));
CREATE TABLE Passport(
ID INT NOT NULL PRIMARY KEY IDENTITY (1,1),
Name NVARCHAR (50),
DOB DATE,
Address NVARCHAR(50));
CREATE TABLE Trip(
ID INT NOT NULL PRIMARY KEY IDENTITY (1,1),
StartDate DATE,
Destination NVARCHAR(25),
Budget INT,
EndDate DATE,
Address NVARCHAR(50));
CREATE TABLE Hotel(
ID INT NOT NULL PRIMARY KEY IDENTITY (1,1),
Name NVARCHAR (50),
CheckInDate DATE,
CheckOutDate DATE,
CostPerNight INT,
PhoneNumber INT,
Address NVARCHAR(50));
